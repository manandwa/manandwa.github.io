def convert_year(year):
    if year >= 11:
        return 'Veteran'
    elif 7 <= year < 11:
        return 'Experienced'
    elif 3 <= year < 7:
        return 'Established'
    elif pd.isnull(year):
        return np.nan
    else:
        return 'New'
